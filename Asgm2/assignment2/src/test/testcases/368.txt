
            func isPrime(number num)
                begin
                    if (num < 2) 
                        return false
                    for i until i * i > num by 1
                            if (num % i = 0) 
                                return false
                    return true
                end

            func main()
                begin
                    var x <- readNumber()
                    if (isPrime(x)) 
                        printString("Prime")
                    else printString("Not Prime")
                end
